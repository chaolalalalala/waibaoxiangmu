{#qklang('Query API')}

{#qklang('Network Data')}

getdifficulty - {#qklang('current mining difficulty')}
gethashrate - {#qklang('estimated hash rate')} (hash/s)
getblockcount - {#qklang('current block height')}
getlasthash - {#qklang('hash of latest block')}

{#qklang('Coin Data')}

blockreward - {#qklang('current block reward')}
coinsupply - {#qklang('total coins mined')}
unminedcoins - {#qklang('total unmined coins')}
runtime - {#qklang('time since first block (secs)')}

{#qklang('Transaction Data')}

txinput/TxHash - {#qklang('total tx input value')}
txoutput/TxHash - {#qklang('total tx output value')}
txfee/TxHash - {#qklang('tx fee value (inputs - outputs)')}
txcount - {#qklang('number of tx,s in blockchain')}

{#qklang('Address Data')}

addressbalance/Address/Confs - {#qklang('balance of address')}
addresslimit/Address/Confs - {#qklang('withdrawal limit of address')}
addresslastseen/Address - {#qklang('block when address last used')}
addresscount - {#qklang('number of non-empty addresses')}

{#qklang('JSON Data')}

getinfo - {#qklang('general information')}
txinfo/TxHash - {#qklang('transaction information')}
addressinfo/Address/Confs - {#qklang('address information')}
blockinfo/BlockHash - {#qklang('block information')}

{#qklang('Usage')}

{#qklang('The following example shows a correct URL for checking the balance of an address, disregarding transactions with less than 3 confirmations (the confirmation argument is always optional, the default value is 1). All other queries which take 1 or more arguments use the same arg1 and arg2 parameter names as shown below.')}

/?q=addressbalance&arg1=JVWEDHJGFDJHWGFDJMOVESAYCOMFDJHGWFDHHGDEW&arg2=3

{#qklang('Or if URL rewriting is active you can use this more friendly format')}:

/q/addressbalance/GJHFHGJFCHJFJHHGFJMOVESAYCOMJGHVFJHVFGH/3
